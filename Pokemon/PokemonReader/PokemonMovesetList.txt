package Pokemon.PokemonBasics.PokemonAllType; // Assuming PokemonType is in this package

public class StringToEnumConverter {

    public static void main(String[] args) {
        String typeString = "FIRE";
        PokemonType pokemonTypeEnum;

        try {
            // Convert String to Enum using valueOf()
            pokemonTypeEnum = PokemonType.valueOf(typeString.toUpperCase()); // Use toUpperCase() for robustness
            System.out.println("Successfully converted string \"" + typeString + "\" to enum: " + pokemonTypeEnum);

            if (pokemonTypeEnum == PokemonType.FIRE) {
                System.out.println("It's a Fire type!");
            }

        } catch (IllegalArgumentException e) {
            System.err.println("Error: No enum constant PokemonType found for string \"" + typeString + "\"");
            // Handle the case where the string doesn't match any enum constant
            pokemonTypeEnum = null; // Or assign a default, or re-throw
        }

        // Example with an invalid string
        String invalidTypeString = "GROUND_TYPE"; // Assuming "GROUND_TYPE" is not an exact constant name
        try {
            PokemonType invalidEnum = PokemonType.valueOf(invalidTypeString);
            System.out.println("Converted: " + invalidEnum);
        } catch (IllegalArgumentException e) {
            System.err.println("Error: No enum constant PokemonType found for string \"" + invalidTypeString + "\"");
        }
    }
}
